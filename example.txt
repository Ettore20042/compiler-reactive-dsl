var speed: int = 0;
var max_speed: int = 100;
var running: bool = true;

task calculateSpeed(base: int, bonus: int) -> int {
    log("Calcolo in corso...");
    return base + bonus;
}

when (speed < max_speed) {
    set speed = calculateSpeed(speed, 10);
    log(speed);
}

while (running == true) {
    set running = false;
    log("Sistema fermato!");
}
